261  void SetDateTime(const RtcDateTime& dt)
262  {
263  // clear the invalid flag
264  uint8_t status = getReg(DS3231_REG_STATUS);
265  status &= ~_BV(DS3231_OSF); // clear the flag
266  setReg(DS3231_REG_STATUS, status);
267 
268  // set the date time
269  _wire.beginTransmission(DS3231_ADDRESS);
270  _wire.write(DS3231_REG_TIMEDATE);
271 
272  _wire.write(Uint8ToBcd(dt.Second()));
273  _wire.write(Uint8ToBcd(dt.Minute()));
274  _wire.write(Uint8ToBcd(dt.Hour())); // 24 hour mode only
275 
276  uint8_t year = dt.Year() - 2000;
277  uint8_t centuryFlag = 0;
278 
279  if (year >= 100)
280  {
281  year -= 100;
282  centuryFlag = _BV(7);
283  }
284 
285  _wire.write(Uint8ToBcd(dt.DayOfWeek()));
286  _wire.write(Uint8ToBcd(dt.Day()));
287  _wire.write(Uint8ToBcd(dt.Month()) | centuryFlag);
288  _wire.write(Uint8ToBcd(year));
289 
290  _wire.endTransmission();
291  }
292  RtcDateTime GetDateTime()
293  {
294  _wire.beginTransmission(DS3231_ADDRESS);
295  _wire.write(DS3231_REG_TIMEDATE);
296  _wire.endTransmission();
297 
298  _wire.requestFrom(DS3231_ADDRESS, DS3231_REG_TIMEDATE_SIZE);
299  uint8_t second = BcdToUint8(_wire.read() & 0x7F);
300  uint8_t minute = BcdToUint8(_wire.read());
301  uint8_t hour = BcdToBin24Hour(_wire.read());
302 
303  _wire.read(); // throwing away day of week as we calculate it
304 
305  uint8_t dayOfMonth = BcdToUint8(_wire.read());
306  uint8_t monthRaw = _wire.read();
307  uint16_t year = BcdToUint8(_wire.read()) + 2000;
308 
309  if (monthRaw & _BV(7)) // century wrap flag
310  {
311  year += 100;
312  }
313  uint8_t month = BcdToUint8(monthRaw & 0x7f);
314 
315 
316  return RtcDateTime(year, month, dayOfMonth, hour, minute, second);
317  }

349  void SetSquareWavePin(DS3231SquareWavePinMode pinMode)
350  {
351  uint8_t creg = getReg(DS3231_REG_CONTROL);
352 
353  // clear all relevant bits to a known "off" state
354  creg &= ~(DS3231_AIEMASK | _BV(DS3231_BBSQW));
355  creg |= _BV(DS3231_INTCN); // set INTCN to disables SQW
356 
357  switch (pinMode)
358  {
359  case DS3231SquareWavePin_ModeNone:
360  break;
361 
362  case DS3231SquareWavePin_ModeBatteryBackup:
363  creg |= _BV(DS3231_BBSQW); // set battery backup flag
364  creg &= ~_BV(DS3231_INTCN); // clear INTCN to enable SQW
365  break;
366 
367  case DS3231SquareWavePin_ModeClock:
368  creg &= ~_BV(DS3231_INTCN); // clear INTCN to enable SQW
369  break;
370 
371  case DS3231SquareWavePin_ModeAlarmOne:
372  creg |= _BV(DS3231_A1IE);
373  break;
374 
375  case DS3231SquareWavePin_ModeAlarmTwo:
376  creg |= _BV(DS3231_A2IE);
377  break;
378 
379  case DS3231SquareWavePin_ModeAlarmBoth:
380  creg |= _BV(DS3231_A1IE) | _BV(DS3231_A2IE);
381  break;
382  }
383 
384  setReg(DS3231_REG_CONTROL, creg);
385  }
