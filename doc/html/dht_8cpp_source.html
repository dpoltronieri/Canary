Canary
The Canary Project is a mobile ambient monitoring device developed in Arduino and written in C++ with a companion app to be developed for Android and iOS. Its goal is to monitor air quality through the measurement of air pollutants, temperature, humidity and sound pollution levels.

69 int8_t dht::read(uint8_t pin){
70  // READ VALUES
71  int8_t result = _readSensor(pin, DHTLIB_DHT_WAKEUP, DHTLIB_DHT_LEADING_ZEROS);
72 
73  // these bits are always zero, masking them reduces errors.
74  bits[0] &= 0x03;
75  bits[2] &= 0x83;
76 
77  // CONVERT AND STORE
78  humidity = (bits[0] * 256 + bits[1]) * 0.1;
79  temperature = ((bits[2] & 0x7F) * 256 + bits[3]) * 0.1;
80  if (bits[2] & 0x80) {
81  // negative temperature
82  temperature = -temperature;
83  }
84 
85  // TEST CHECKSUM
86  uint8_t sum = bits[0] + bits[1] + bits[2] + bits[3];
87  if (bits[4] != sum) {
88  return DHTLIB_ERROR_CHECKSUM;
89  }
90  return result;

98 int8_t dht::_readSensor(uint8_t pin, uint8_t wakeupDelay, uint8_t leadingZeroBits){
99  // INIT BUFFERVAR TO RECEIVE DATA
100  uint8_t mask = 128;
101  uint8_t idx = 0;
102 
103  uint8_t data = 0;
104  uint8_t state = LOW;
105  uint8_t pstate = LOW;
106  uint16_t zeroLoop = DHTLIB_TIMEOUT;
107  uint16_t delta = 0;
108 
109  leadingZeroBits = 40 - leadingZeroBits; // reverse counting...
110 
111  // replace digitalRead() with Direct Port Reads.
112  // reduces footprint ~100 bytes => portability issue?
113  // direct port read is about 3x faster
114  uint8_t bit = digitalPinToBitMask(pin);
115  uint8_t port = digitalPinToPort(pin);
116  volatile uint8_t * PIR = portInputRegister(port);
117 
118  // REQUEST SAMPLE
119  pinMode(pin, OUTPUT);
120  digitalWrite(pin, LOW); // T-be
121  delayMicroseconds(wakeupDelay * 1000UL);
122  digitalWrite(pin, HIGH); // T-go
123  pinMode(pin, INPUT);
124 
125  uint16_t loopCount = DHTLIB_TIMEOUT * 2; // 200uSec max
126  // while(digitalRead(pin) == HIGH)
127  while ((*PIR & bit) != LOW) {
128  if (--loopCount == 0) return DHTLIB_ERROR_CONNECT;
129  }
130 
131  // GET ACKNOWLEDGE or TIMEOUT
132  loopCount = DHTLIB_TIMEOUT;
133  // while(digitalRead(pin) == LOW)
134  while ((*PIR & bit) == LOW) {
135  // T-rel
136  if (--loopCount == 0) return DHTLIB_ERROR_ACK_L;
137  }
138 
139  loopCount = DHTLIB_TIMEOUT;
140  // while(digitalRead(pin) == HIGH)
141  while ((*PIR & bit) != LOW) {
142  // T-reh
143  if (--loopCount == 0) return DHTLIB_ERROR_ACK_H;
144  }
145 
146  loopCount = DHTLIB_TIMEOUT;
147 
148  // READ THE OUTPUT - 40 BITS => 5 BYTES
149  for (uint8_t i = 40; i != 0;) {
150  // WAIT FOR FALLING EDGE
151  state = (*PIR & bit);
152  if (state == LOW && pstate != LOW) {
153  if (i > leadingZeroBits) {
154  // DHT22 first 6 bits are all zero !! DHT11 only 1
155  zeroLoop = min(zeroLoop, loopCount);
156  delta = (DHTLIB_TIMEOUT - zeroLoop) / 4;
157  } else if (loopCount <= (zeroLoop - delta) ) {
158  // long -> one
159  data |= mask;
160  }
161  mask >>= 1;
162  if (mask == 0) {
163  // next byte
164  mask = 128;
165  bits[idx] = data;
166  idx++;
167  data = 0;
168  }
169  // next bit
170  --i;
171 
172  // reset timeout flag
173  loopCount = DHTLIB_TIMEOUT;
174  }
175  pstate = state;
176  // Check timeout
177  if (--loopCount == 0) {
178  return DHTLIB_ERROR_TIMEOUT;
179  }
180  }
181  pinMode(pin, OUTPUT);
182  digitalWrite(pin, HIGH);
183 
184  return DHTLIB_OK;
185 } // _readSensor
