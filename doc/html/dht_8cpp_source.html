94 float DHT::computeHeatIndex(float temperature, float percentHumidity, bool isFahrenheit){
95  // Using both Rothfusz and Steadman's equations
96  // http://www.wpc.ncep.noaa.gov/html/heatindex_equation.shtml
97  float hi;
98 
99  if (!isFahrenheit)
100  temperature = convertCtoF(temperature);
101 
102  hi = 0.5 * (temperature + 61.0 + ((temperature - 68.0) * 1.2) + (percentHumidity * 0.094));
103 
104  if (hi > 79) {
105  hi = -42.379
106  + 2.04901523 * temperature
107  + 10.14333127 * percentHumidity
108  + -0.22475541 * temperature * percentHumidity
109  + -0.00683783 * pow(temperature, 2)
110  + -0.05481717 * pow(percentHumidity, 2)
111  + 0.00122874 * pow(temperature, 2) * percentHumidity
112  + 0.00085282 * temperature * pow(percentHumidity, 2)
113  + -0.00000199 * pow(temperature, 2) * pow(percentHumidity, 2);
114 
115  if ((percentHumidity < 13) && (temperature >= 80.0) && (temperature <= 112.0))
116  hi -= ((13.0 - percentHumidity) * 0.25) * sqrt((17.0 - abs(temperature - 95.0)) * 0.05882);
117 
118  else if ((percentHumidity > 85.0) && (temperature >= 80.0) && (temperature <= 87.0))
119  hi += ((percentHumidity - 85.0) * 0.1) * ((87.0 - temperature) * 0.2);
120  }
121 
122  return isFahrenheit ? hi : convertFtoC(hi);
123 }

125 boolean DHT::read(bool force){
126  // Check if sensor was read less than two seconds ago and return early
127  // to use last reading.
128  uint32_t currenttime = millis();
129 
130  if (!force && ((currenttime - _lastreadtime) < 2000)) {
131  return _lastresult; // return last correct measurement
132  }
133  _lastreadtime = currenttime;
134 
135  // Reset 40 bits of received data to zero.
136  data[0] = data[1] = data[2] = data[3] = data[4] = 0;
137 
138  // Send start signal. See DHT datasheet for full signal diagram:
139  // http://www.adafruit.com/datasheets/Digital%20humidity%20and%20temperature%20sensor%20AM2302.pdf
140 
141  // Go into high impedence state to let pull-up raise data line level and
142  // start the reading process.
143  digitalWrite(_pin, HIGH);
144  delay(250);
145 
146  // First set data line low for 20 milliseconds.
147  pinMode(_pin, OUTPUT);
148  digitalWrite(_pin, LOW);
149  delay(20);
150 
151  uint32_t cycles[80];
152  {
153  // Turn off interrupts temporarily because the next sections are timing critical
154  // and we don't want any interruptions.
155  InterruptLock lock;
156 
157  // End the start signal by setting data line high for 40 microseconds.
158  digitalWrite(_pin, HIGH);
159  delayMicroseconds(40);
160 
161  // Now start reading the data line to get the value from the DHT sensor.
162  pinMode(_pin, INPUT_PULLUP);
163  delayMicroseconds(10); // Delay a bit to let sensor pull data line low.
164 
165  // First expect a low signal for ~80 microseconds followed by a high signal
166  // for ~80 microseconds again.
167  if (expectPulse(LOW) == 0) {
168  DEBUG_PRINTLN(F("Timeout waiting for start signal low pulse."));
169  _lastresult = false;
170  return _lastresult;
171  }
172  if (expectPulse(HIGH) == 0) {
173  DEBUG_PRINTLN(F("Timeout waiting for start signal high pulse."));
174  _lastresult = false;
175  return _lastresult;
176  }
177 
178  // Now read the 40 bits sent by the sensor. Each bit is sent as a 50
179  // microsecond low pulse followed by a variable length high pulse. If the
180  // high pulse is ~28 microseconds then it's a 0 and if it's ~70 microseconds
181  // then it's a 1. We measure the cycle count of the initial 50us low pulse
182  // and use that to compare to the cycle count of the high pulse to determine
183  // if the bit is a 0 (high state cycle count < low state cycle count), or a
184  // 1 (high state cycle count > low state cycle count). Note that for speed all
185  // the pulses are read into a array and then examined in a later step.
186  for (int i = 0; i < 80; i += 2) {
187  cycles[i] = expectPulse(LOW);
188  cycles[i + 1] = expectPulse(HIGH);
189  }
190  } // Timing critical code is now complete.
191 
192  // Inspect pulses and determine which ones are 0 (high state cycle count < low
193  // state cycle count), or 1 (high state cycle count > low state cycle count).
194  for (int i = 0; i < 40; ++i) {
195  uint32_t lowCycles = cycles[2 * i];
196  uint32_t highCycles = cycles[2 * i + 1];
197  if ((lowCycles == 0) || (highCycles == 0)) {
198  DEBUG_PRINTLN(F("Timeout waiting for pulse."));
199  _lastresult = false;
200  return _lastresult;
201  }
202  data[i / 8] <<= 1;
203  // Now compare the low and high cycle times to see if the bit is a 0 or 1.
204  if (highCycles > lowCycles) {
205  // High cycles are greater than 50us low cycle count, must be a 1.
206  data[i / 8] |= 1;
207  }
208  // Else high cycles are less than (or equal to, a weird case) the 50us low
209  // cycle count so this must be a zero. Nothing needs to be changed in the
210  // stored data.
211  }
212 
213  DEBUG_PRINTLN(F("Received:"));
214  DEBUG_PRINT(data[0], HEX);
215  DEBUG_PRINT(F(", "));
216  DEBUG_PRINT(data[1], HEX);
217  DEBUG_PRINT(F(", "));
218  DEBUG_PRINT(data[2], HEX);
219  DEBUG_PRINT(F(", "));
220  DEBUG_PRINT(data[3], HEX);
221  DEBUG_PRINT(F(", "));
222  DEBUG_PRINT(data[4], HEX);
223  DEBUG_PRINT(F(" =? "));
224  DEBUG_PRINTLN((data[0] + data[1] + data[2] + data[3]) & 0xFF, HEX);
225 
226  // Check we read 40 bits and that the checksum matches.
227  if (data[4] == ((data[0] + data[1] + data[2] + data[3]) & 0xFF)) {
228  _lastresult = true;
229  return _lastresult;
230  } else {
231  DEBUG_PRINTLN(F("Checksum failure!"));
232  _lastresult = false;
233  return _lastresult;
234  }
235 } // read

237 // Expect the signal line to be at the specified level for a period of time and
238 // return a count of loop cycles spent at that level (this cycle count can be
239 // used to compare the relative time of two pulses). If more than a millisecond
240 // ellapses without the level changing then the call fails with a 0 response.
241 // This is adapted from Arduino's pulseInLong function (which is only available
242 // in the very latest IDE versions):
243 // https://github.com/arduino/Arduino/blob/master/hardware/arduino/avr/cores/arduino/wiring_pulse.c
244 uint32_t DHT::expectPulse(bool level){
245  uint32_t count = 0;
246 
247  // On AVR platforms use direct GPIO port access as it's much faster and better
248  // for catching pulses that are 10's of microseconds in length:
249  #ifdef __AVR
250  uint8_t portState = level ? _bit : 0;
251  while ((*portInputRegister(_port) & _bit) == portState) {
252  if (count++ >= _maxcycles) {
253  return 0; // Exceeded timeout, fail.
254  }
255  }
256  // Otherwise fall back to using digitalRead (this seems to be necessary on ESP8266
257  // right now, perhaps bugs in direct port access functions?).
258  #else
259  while (digitalRead(_pin) == level) {
260  if (count++ >= _maxcycles) {
261  return 0; // Exceeded timeout, fail.
262  }
263  }
264  #endif // ifdef __AVR
265 
266  return count;
267 }
